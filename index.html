Sub DeleteBlankRowsInColumnA(ByVal ws As Worksheet)

    Dim lastRow As Long
    Dim i As Long

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    ' 下から上へ削除
    For i = lastRow To 1 Step -1
        If Trim(ws.Cells(i, 1).Value) = "" Then
            ws.Rows(i).Delete
        End If
    Next i

End Sub
