Sub DeleteRows(ByVal ws As Worksheet,)

    Dim lastRow As Long
    Dim i As Long

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    ' 下から上へ削除
    For i = lastRow To 1 Step -1
        If Trim(ws.Cells(i, 1).Value) = "" Then
            ws.Rows(i).Delete
        End If
    Next i

End Sub

問題ありなしを調べてTrueFalseをかえすが、ByRefでメッセージを変える


==================================================================
Function 時間問題あり(ByRef msg As String) As Boolean
    msg = ""                ' メッセージ初期化
    時間問題あり = False   ' 初期値

    If Now > TimeValue("15:00:00") Then
        msg = msg & "15時以降です。" & vbCrLf
    End If

    If Now < TimeValue("20:00:00") Then
        msg = msg & "20時以前です。" & vbCrLf
    End If

    If msg <> "" Then
        時間問題あり = True
    End If
End Function


Sub 実行()
    Dim message As String

    If 時間問題あり(message) Then
        MsgBox message & vbCrLf & "時間問題ありです。処理を終了します。"
        Exit Sub
    End If
End Sub
