Private Function DeleteRowsByValue(ByVal ws As Worksheet, ByVal target As String, ByVal column As Long) As Long
    'column列の値がtargetの行を削除する
    '削除した行数を返す
    Dim v As Variant
    Dim i As Long
    Dim lastRow As Long
    Dim deleteRowRng As Range

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    v = ws.Range(ws.Cells(1, column), ws.Cells(lastRow, column)).value

    For i = 1 To lastRow
        If v(i, 1) = target Then
            If deleteRowRng Is Nothing Then
                Set deleteRowRng = ws.Rows(i)
            Else
                Set deleteRowRng = Union(deleteRowRng, ws.Rows(i))
            End If
        End If
    Next i

    If Not deleteRowRng Is Nothing Then
        DeleteRowsByValue = deleteRowRng.Rows.Count
        deleteRowRng.Delete
    Else
        DeleteRowsByValue = 0
    End If
End Function

Private Function 時間問題あり(ByRef msg As String) As Boolean
    msg = ""                ' メッセージ初期化
    時間問題あり = False   ' 初期値

    If Now > TimeValue("15:00:00") Then
        msg = msg & "15時以降です。" & vbCrLf
    End If

    If Now < TimeValue("20:00:00") Then
        msg = msg & "20時以前です。" & vbCrLf
    End If

    If msg <> "" Then
        時間問題あり = True
    End If
End Function


Sub 実行()
    Dim message As String

    If 時間問題あり(message) Then
        MsgBox message & vbCrLf & "時間問題ありです。処理を終了します。"
        Exit Sub
    End If
End Sub

Private Function CollectionToArray(ByVal collection As collection, ByVal startIndex As Long) As Variant()
    'コレクションを配列に変換する
    Dim arr() As Variant
    Dim i As Long

    ' 配列を開始番号に合わせて確保
    ReDim arr(startIndex To startIndex + collection.Count - 1)

    ' Collection → 配列へコピー
    For i = 1 To collection.Count
        arr(startIndex + i - 1) = collection.Item(i)
    Next i

    CollectionToArray = arr
End Function

Private Function IsWorkbookLocked(ByVal filePath As String) As Boolean
    'ブックが開ひらかれているか調べる
    Dim fileNum As Integer

    On Error GoTo Locked

    fileNum = FreeFile

    ' 追記モード（書き込みロック）で開けるか試す
    Open filePath For Binary Access Write Lock Read Write As #fileNum

    ' 開けた → ロックされていない
    Close #fileNum
    IsWorkbookLocked = False
    Exit Function

Locked:
    ' 開けない → 誰かが開いている（編集中）
    IsWorkbookLocked = True
End Function

Private Function GetLastRow(ByVal ws As Worksheet) As Long
    '最終行を取得
    Dim lastRow As Long
    Dim col_i As Long
    Dim r As Long

    lastRow = 0

    ' 実務で使う範囲だけ（1～100列）
    For col_i = 1 To 100
        r = ws.Cells(ws.Rows.Count, col_i).End(xlUp).Row
        If r > lastRow Then lastRow = r
    Next col_i

    GetLastRow = lastRow
End Function

Private Function HasDuplicate(ByVal ws As Worksheet, ByVal checkColumn As Integer) As Boolean
    '重複があるか調べる
    Dim dic As Object
    Dim v As Variant
    Dim i As Long
    Dim lastRow As Long
    
    HasDuplicate = False

    Set dic = CreateObject("Scripting.Dictionary")
    
    lastRow = ws.Cells(ws.Rows.Count, checkColumn).End(xlUp).Row
    v = ws.Range(ws.Cells(1, checkColumn), ws.Cells(lastRow, checkColumn)).value

    For i = 1 To UBound(v)
        If dic.Exists(v(i, 1)) Then
            HasDuplicate = True
            Exit Function
        Else
            dic.Add v(i, 1), i
        End If
    Next i
    
End Function

３列を結合した値に重複があるか


==================================================================
Private Function HasDuplicate(ByVal ws As Worksheet, ByVal checkColumn1 As Integer, ByVal checkColumn2 As Integer, ByVal checkColumn3 As Integer) As Boolean
    '重複があるか調べる
    Dim dic As Object
    Dim v As Variant
    Dim i As Long
    Dim lastRow As Long
    Dim maxColumn As Long
    Dim minColumn As Long
    Dim key As String
    
    HasDuplicate = False

    Set dic = CreateObject("Scripting.Dictionary")
    
    lastRow = ws.Cells(ws.Rows.Count, checkColumn1).End(xlUp).Row
    maxColumn = WorksheetFunction.Max(checkColumn1, checkColumn2, checkColumn3)
    minColumn = WorksheetFunction.Min(checkColumn1, checkColumn2, checkColumn3)
    
    v = ws.Range(ws.Cells(1, minColumn), ws.Cells(lastRow, maxColumn)).value

    For i = 1 To UBound(v)
        key = CStr(v(i, checkColumn1)) & "|" & CStr(v(i, checkColumn2)) & "|" & CStr(v(i, checkColumn3))
        If dic.Exists(key) Then
            HasDuplicate = True
            Exit Function
        Else
            dic.Add key, i
        End If
    Next i
    
End Function
